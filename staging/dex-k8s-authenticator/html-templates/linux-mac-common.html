{{ define "linux-mac-common" }} {{ if .IDPCaURI }}

Copy IDP CA Certificate From URL

Copy this CA Certificate and download it to your .kube directory

curl --create-dirs -s {{ .IDPCaURI }} -o ${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/idp-ca.crt
{{ end }} {{ if .IDPCaPem }}

Copy IDP CA Certificate From PEM

Put the CA Certificate into your .kube directory

mkdir -p ${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/ && cat << EOF > ${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/idp-ca.crt
{{ .IDPCaPem }}
EOF
{{ end }} {{ if .K8sCaURI }}

Copy Kubernetes CA Certificate From URL

Copy this CA Certificate and download it to your .kube directory

curl --create-dirs -s {{ .K8sCaURI }} -o ${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/k8s-ca.crt
{{ end }} {{ if .K8sCaPem }}

Copy Kubernetes CA Certificate From PEM

Put the CA Certificate into your .kube directory

mkdir -p ${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/ && cat << EOF > ${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/k8s-ca.crt
{{ .K8sCaPem }}
EOF
{{ end }}

Run configuration commands

These commands will update ~/.kube/config

kubectl config set-cluster {{ .Username }}-{{ .ClusterHostname }} \
  {{- if or .K8sCaPem .K8sCaURI }}
    --certificate-authority=${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname}}/k8s-ca.crt \
  {{- end }}
    --server={{ .K8sMasterURI }}
kubectl config set-credentials {{ .Username }}-{{ .ClusterHostname }} \
    --token={{ .IDToken }}
  {{- if or (.IDPCaURI) (.IDPCaPem) }} \
    --auth-provider-arg=idp-certificate-authority=${HOME}/.kube/certs/{{ .Username }}-{{ .ClusterHostname }}/idp-ca.crt
  {{- end }}
kubectl config set-context {{ .Username }}-{{ .ClusterHostname }} \
    --cluster={{ .Username }}-{{ .ClusterHostname }} \
    --user={{ .Username }}-{{.ClusterHostname }}
kubectl config use-context {{ .Username }}-{{ .ClusterHostname}}
{{ end }}
